

CREATE TABLE USUARIOS
(
ID_USER INTEGER NOT NULL GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1),
USERNAME VARCHAR(30) NOT NULL,
PASSWORD VARCHAR(30),
CONSTRAINT primary_key PRIMARY KEY (ID_USER)
) ;

CREATE TABLE BASES(
ID_DB INTEGER NOT NULL GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1),
USERID INTEGER NOT NULL,
DBNAME VARCHAR(30),
CONSTRAINT primary_key_DB PRIMARY KEY (ID_DB),
CONSTRAINT userid_ref FOREIGN KEY (USERID) REFERENCES USUARIOS(ID_USER)
) ;
